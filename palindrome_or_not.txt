#include<iostream>
using namespace std;
int main()
{
	string str="aba";
	bool s;
	char arr[3];
	int j=0;
	for(int i=str.length()-1;i>=0;i--)
	{
		arr[j]=str[i];
		j++;
	}
	
	for(int i=0;i<str.length();i++)
	{
		if(str[i]==arr[i])
		{
			s=true;
		}
		else
		{
			s=false;
			break;
		}
	}
	if(s==true)
	{
		cout<<"exact match";
	}
	else{
		cout<<"not match";
	}
}
